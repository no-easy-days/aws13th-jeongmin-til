- 클래스 생성 및 활용
```
class EC2Instance:
    def __init__(self, instance_id, name, status = "stopped"):
        self.instance_id = instance_id
        self.name = name
        self.status = status

    def start(self):
        self.status = "running"

    def stop(self):
        self.status = "stopped"

class EC2Manager:
    def __init__(self):
        self.instances = []

    def add_instance(self,instance):
        self.instances.append(instance)
        print(f"추가 인스턴스 : {self.instances}")

    def start_all(self):
        for i in self.instances:
            i.start()

    def stop_all(self):
        for i in self.instances:
            i.stop()

    def get_running_count(self):
        count = 0
        for instance in self.instances:
            if instance.status == "running":
                count += 1
        return count

web = EC2Instance("i-001", "web-server")
db = EC2Instance("i-002", "db-server")
cache = EC2Instance("i-003", "cache-server")

manager = EC2Manager()
manager.add_instance(web)
manager.add_instance(db)
manager.add_instance(cache)

manager.start_all()
print(manager.get_running_count()) 

db.stop()
print(manager.get_running_count()) 
```
