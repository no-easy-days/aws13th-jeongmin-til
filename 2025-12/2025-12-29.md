#### 파이썬 자료형
- 파이썬은 **동적 타이핑 언어**
- 변수에 자료형을 지정하지 않음
- **값이 자료형을 가진다**
  
```
# 정수형은 소수점이 없는 숫자
a = 10

# 파이썬의 int는 크기 제한이 없다
b = 10 ** 100

# 실수형은 소수점을 포함한 숫자를 표현한다
pi = 3.14

# 부동소수점 오차가 발생할 수 있다 
result = 0.1 + 0.2

# decimal 모듈은 10진수 기반으로 정확한 계산을 제공한다
from decimal import Decimal

# 문자열로 Decimal 객체를 생성해야 정확하다
result_decimal = Decimal("0.1") + Decimal("0.2")

# 문자열은 문자들의 나열 (인덱스로 접근이 가능)
text = "python"

# 문자열은 불변(immutable) 객체이다
new_text = text + " language"

# 불리언은 참(True) 또는 거짓(False)을 표현한다
is_valid = True

# 비교 연산의 결과는 bool 타입이다
check = (10 > 5)

# None은 값이 없음을 의미한다
data = None

# None 비교는 is 연산자를 사용한다
is_empty = (data is None)

# 리스트는 여러 값을 순서대로 저장한다
nums = [1, 2, 3]

# 리스트는 가변(mutable) 객체이다
nums.append(4)

# 튜플은 여러 값을 묶어 표현한다
point = (10, 20)

# 튜플은 불변(immutable) 객체이다
same_point = point + (30,)

# 딕셔너리는 key-value 구조이다
user = {"name": "Tom", "age": 20}

# 키를 이용해 값을 추가하거나 수정할 수 있다
user["age"] = 21

# SET은 중복을 허용하지 않는 자료형이다
nums = {1, 2, 3}

# 집합은 가변 객체이며 요소를 추가할 수 있다
nums.add(4)

```
#### 변수
```
# 재 할당
score = 80
print(score)  # 출력: 80
score = 95    # 새로운 값으로 변경
print(score)  # 출력: 95

# 같은 값 다중 할당
x = y = z = 0
print(x, y, z)  # 출력: 0 0 0

# 다른 값 다중 할당
name, age, city = "철수", 25, "서울"
print(name)  # 출력: 철수
print(age)   # 출력: 25
print(city)  # 출력: 서울

# 값 교환 
a = 10
b = 20
# 파이썬 방식: 한 줄로 교환
a, b = b, a
print(a)  # 출력: 20
print(b)  # 출력: 10
# 값을 교환 하기 위한 추가적인 변수 필요 X 
```
